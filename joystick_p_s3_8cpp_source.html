148 arm_event JoystickPS3::Read() {
149  arm_event newMovement = {};
150  SDL_Event event;
151  // Catch event
152  SDL_PollEvent(&event);
153  // if event is press button or modify axis
154  if (event.type == SDL_JOYAXISMOTION || event.type == SDL_JOYBUTTONDOWN ||
155  event.type == SDL_JOYBUTTONUP || event.type == SDL_JOYHATMOTION) {
156  _print_event(&event);
157  _set_movement(&event);
158  }
159  newMovement = _is_new_event();
160  // Handle hot unplugging
161  if (event.type == SDL_JOYDEVICEADDED && isOpen == false) {
162  Open();
163  }
164  if (event.type == SDL_JOYDEVICEREMOVED && isOpen == true) {
165  hotUnplug = true;
166  Close();
167  }
168  return newMovement;
169 }
